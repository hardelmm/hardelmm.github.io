107    public boolean processTranscodefree(String sourcefile,String VideoBitrate,String resolution,String AudioBitrate,String AudioSampleRate,String CreateFileName,String CreateFileFormat) throws Exception{  
108        File diretory = new File("");
109        String currPath = diretory.getAbsolutePath();
110        String outputPath = currPath + "\\output\\";  
111        String ffmpegPath = currPath + "\\FFmpeg\\bin\\";  
112        if (!checkfile(sourcefile)) {
113            System.out.println(sourcefile + " is not file");  
114        }
115        // 文件命名  
116        //Calendar c = Calendar.getInstance(); 
117        List<String> commend = new ArrayList<String>();
118        commend.add(ffmpegPath+"ffmpeg.exe"); 
119        //commend.add("F:\\java\\TranscodeGUI\\ffmpeg.exe"); 
120        commend.add("-i");
121        commend.add(sourcefile);
122        commend.add("-ab");
123        commend.add("-vcodec");
124        commend.add("h264");
125        //commend.add("");
126        commend.add("-b:v");
127        commend.add(VideoBitrate);
128        commend.add("-r");
129        commend.add("25");
130        commend.add("-s");
131        commend.add("resolution");
132        commend.add("-pix_fmt");
133        commend.add("yuv420p");
134        commend.add("-g");
135        commend.add("33");
136        commend.add("-acodec");
137        commend.add("aac");
138        commend.add("-b:a");
139        commend.add(AudioBitrate);
140        commend.add("-ac");
141        commend.add("2");
142        commend.add("-strict");
143        commend.add("-2");
144        commend.add("-f");
145        commend.add(CreateFileFormat);
146        commend.add("-y");
147        commend.add(outputPath+CreateFileName+"."+CreateFileFormat);
148        try {
149            ProcessBuilder builder = new ProcessBuilder();  
150            builder.command(commend);  
151            Process proc = builder.start();
152            InputStream stderr = proc.getErrorStream();
153            InputStreamReader isr = new InputStreamReader(stderr);
154            BufferedReader br = new BufferedReader(isr);
155            String line = null;
156            while ( (line = br.readLine()) != null)
157                System.out.println(line);
158
159            int exitVal = proc.waitFor();
160            System.out.println("Process exitValue: " + exitVal);
161        } catch (Throwable t) {
162            t.printStackTrace();
163            return false;
164        }
165        return true;
166    }
